  <script>
    const canvas = document.getElementById("gameCanvas");

    const spriteFiles = {

const BOOGER_SIZE = 32;
const BOOGER_ANIM = {
    moveRight: { row: 5, frames: 4 },
    moveLeft:  { row: 6, frames: 4 },
    hitRight:  { row: 13, frames: 2 },
    hitLeft:   { row: 15, frames: 2 },

};

    function assetLoaded() {
      loadedAssets += 1;
      if (loadedAssets >= totalAssets) {
        startGame();
      }
    }

    Object.entries(spriteFiles).forEach(([key, src]) => {
      const img = new Image();
      img.onload = assetLoaded;
      img.onerror = assetLoaded;
      img.src = src;
      sprites[key] = img;

    const GRAVITY = 0.6;

    let mobs;

const FLAG_FRAME_DURATION = 150; // Milliseconds per frame

      player = {

    ];

goal = {
  x: 1330, 
  y: 92, 
  w: 44, 
  h: 58,

  hitboxOffsetX: 14, 
  hitboxOffsetY: 8,
  frameCount: 5,
  frameIndex: 0,
  // This is the only distance that matters
  frameWidth: 60,
  active: false
};
      score = 0;
      keys = {};

mobs = [
  { x: 300, y: 285, w: 50, h: 40, dx: 1.5, health: 3, frame: 0, facing: 1, wasHit: false, isDead: false, deathTimer: 0},
  { x: 500, y: 285, w: 50, h: 40, dx: -1.5, health: 3, frame: 0, facing: -1, wasHit: false, isDead: false, deathTimer: 0}
];

    function startGame() {
      resetGame();
      running = true;
      lastTime = performance.now();
      overlay.style.display = "none";
      if (animationFrameId) cancelAnimationFrame(animationFrameId);
      animationFrameId = requestAnimationFrame(loop);
    }

    function showOverlay(title, text, hint) {
      running = false;
      if (animationFrameId) cancelAnimationFrame(animationFrameId);
      overlay.style.display = "flex";
      panelTitle.textContent = title;
      panelText.textContent = text;
      panelHint.textContent = hint || "Controls: Arrow keys or A/D to move, Space/Up/W to jump.";
      startButton.textContent = title === "You Win!" ? "Play Again" : "Restart";
    }

    function finishGame(title, text) {
    if (title === "You Win!") {
        // Show the secret word popup instead of the standard overlay
        const popup = document.getElementById("goalPopup");
        popup.style.display = "flex";
        document.getElementById("passwordInput").focus();
    } else {
        // Default behavior for "Try Again"
        showOverlay(title, text);
    }
}

    function loop(time) {
      if (!running) return;
      const delta = Math.min(2, (time - lastTime) / 16.67);
      lastTime = time;
      update(delta);
      render();
      animationFrameId = requestAnimationFrame(loop);
    }

    function update(delta) {
// Inside update(delta)
const moveInput = (keys["ArrowRight"] || keys["KeyD"] ? 1 : 0) - (keys["ArrowLeft"] || keys["KeyA"] ? 1 : 0);

if (moveInput !== 0) {
  player.dx = moveInput * MOVE_SPEED; // Remove * delta here for smoother horizontal movement
  player.facing = moveInput;
} else {
  player.dx *= FRICTION; // This slows the player down gradually
}

      if ((keys["Space"] || keys["ArrowUp"] || keys["KeyW"]) && player.onGround) {
        player.dy = JUMP_FORCE;
        player.onGround = false;
      }

      player.dy += GRAVITY * delta;
      player.x += player.dx;
      player.y += player.dy;

      let landed = false;
      const playerHitboxX = player.x + (player.w - player.hitboxW) / 2;
      
      platforms.forEach((platform) => {
        if (
          playerHitboxX < platform.x + platform.w &&
          playerHitboxX + player.hitboxW > platform.x &&
          player.y + player.hitboxH >= platform.y &&
          player.y + player.hitboxH <= platform.y + 20
        ) {
          if (player.dy >= 0) {
            player.y = platform.y - player.hitboxH;
            player.dy = 0;
            player.onGround = true;
            landed = true;
          }
        }
      });

      if (!landed) player.onGround = false;

     const hitboxPadding = (player.w - player.hitboxW) / 2;

if (player.x + hitboxPadding < 0) {
  player.x = -hitboxPadding;
}
if (player.x + hitboxPadding + player.hitboxW > WORLD_WIDTH) {
  player.x = WORLD_WIDTH - player.hitboxW - hitboxPadding;
}
      if (player.y > canvas.height + 80) {
        finishGame("Try Again", "You fell off the platforms. Jump carefully and reach the flag.");
        return;
      }

       // Add to the end of your update function
mobs.forEach(mob => {
  if (mob.health <= 0) return; // Only process if alive
  
  mob.x += mob.dx * delta;
  if (mob.x < 100 || mob.x > 700) mob.dx *= -1;

  // Check for hit
  if (player.isAttacking) {
    const comboFrame = Math.floor(player.playerFrameIndex / 4);
    
    // Only hit during frames 2, 3, 4 (active swing)
    // AND check if they haven't been hit by this specific swing yet
    if (comboFrame >= 2 && comboFrame <= 4 && !mob.wasHit) {
      
      const hitBoxX = player.facing === 1 ? player.x + 40 : player.x - 40;
      
if (Math.abs(mob.x - hitBoxX) < 60) {
  mob.health -= 1;
  mob.wasHit = true;

  // Pick a random letter from the pool
  if (letterPool.length > 0) {
    const randomIndex = Math.floor(Math.random() * letterPool.length);
    const letter = letterPool.splice(randomIndex, 1)[0]; 
    collectedLetters.push(letter);
    
    // This displays the letter floating over the mob
    spawnDamageText(mob.x, mob.y - 20, letter.toUpperCase());
  }
  
  if (mob.health <= 0) {
    score += 100;
  }
}
    }
  } else {
    // Reset the hit flag when the player finishes the attack
    mob.wasHit = false;
  }

  if (mob.health <= 0 && !mob.isDead) {
    mob.isDead = true;
    mob.frame = 0; // Reset frame to start of death animation
    
  }

});
  mobs = mobs.filter(m => !m.removed);    

    flagTimer += delta * 16.67; // Convert frames to milliseconds
if (flagTimer >= FLAG_FRAME_DURATION) {
    goal.frameIndex = (goal.frameIndex + 1) % goal.frameCount;
    flagTimer = 0;
}

const goalHitboxX = goal.x + goal.hitboxOffsetX;
const goalHitboxY = goal.y + goal.hitboxOffsetY;

// Add the 'if (goal.active)' check here
if (goal.active) { 
    if (
      player.x < goalHitboxX + goal.hitboxW &&
      player.x + player.hitboxW > goalHitboxX &&
      player.y < goalHitboxY + goal.hitboxH &&

if (!goal.active && letterPool.length === 0) {
  goal.active = true;
  console.log("Goal Unlocked!");
  spawnDamageText(goal.x, goal.y - 20, "UNLOCKED!");

function render() {
    // 1. Calculate camera
    camera.x = Math.max(0, Math.min(player.x - canvas.width / 2, WORLD_WIDTH - canvas.width));

    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // --- A. Draw Static Background (No Translation) ---
    const sky = ctx.createLinearGradient(0, 0, 0, canvas.height);
    sky.addColorStop(0, "#8fd4ff");
    sky.addColorStop(1, "#dff6ff");
    ctx.fillStyle = sky;
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // --- B. Start Camera Translation ---
    ctx.save();
    ctx.translate(-camera.x, 0);

// 1. Draw the Base Dirt (Brown)
ctx.fillStyle = "#7a4b29";
ctx.fillRect(0, 350, WORLD_WIDTH, 50);

// 2. Draw the Dark Brown Trim (Line at the very top of the dirt)
ctx.fillStyle = "#5b331a";
ctx.fillRect(0, 350, WORLD_WIDTH, 4);

// 3. Draw the Green Grass (Layer above the trim)
ctx.fillStyle = "#8ddf7d";
ctx.fillRect(0, 340, WORLD_WIDTH, 10);

// 4. Now draw the other platforms from the array
platforms.forEach((platform) => {
    // Skip drawing the floor as a platform here to avoid double-drawing 
    // OR just draw them all as standard platforms
    if (platform.y === 350 && platform.x === 0) return; 

    ctx.fillStyle = "#7a4b29";
    ctx.fillRect(platform.x, platform.y, platform.w, platform.h);
    ctx.fillStyle = "#5b331a";
    ctx.fillRect(platform.x, platform.y - 4, platform.w, 4);
});
    // Draw Mobs
    mobs.forEach(m => {
        if (m.health >= 0) {
            m.facing = m.dx > 0 ? 1 : -1;
            drawBooger(m);
        }
    });

    // Draw Goal
    ctx.drawImage(goalImage, goal.frameIndex * 60, 0, 32, 50, goal.x, goal.y, goal.w, goal.h);
    if (!goal.active) {
        ctx.font = "24px Arial";
        ctx.textAlign = "center";
        ctx.fillText("🔒", goal.x + goal.w / 2, goal.y - 10);
    }

    // Draw Player
    drawPlayer();

    ctx.restore(); // --- C. End Camera Translation ---

    // --- D. Draw UI (No Translation - Pinned to Screen) ---
    for (let i = floatingTexts.length - 1; i >= 0; i--) {
        let ft = floatingTexts[i];
        ft.y -= 0.5; 
        ft.timer -= 1;
        ctx.save();
        ctx.globalAlpha = ft.timer / 60;
        ctx.fillStyle = "#FFD700";
        ctx.font = ft.text === "UNLOCKED!" ? "bold 16px Arial" : "bold 30px Arial";
        ctx.textAlign = "center";
        // Calculate position relative to camera for floating text
        ctx.fillText(ft.text, ft.x - camera.x, ft.y);
        ctx.restore();
        if (ft.timer <= 0) floatingTexts.splice(i, 1);
    }

    ctx.fillStyle = "rgba(255,255,255,0.85)";
    ctx.font = "16px Segoe UI";
    ctx.textAlign = "left";
    ctx.fillText(`Score: ${score}`, 30, 28);
}

 function drawPlayer() {
  let img = sprites.idle;
  let frame = 0;
  let frameCount = 10;
  
  const isMoving = Math.abs(player.dx) > 0.5;
  const isJumping = player.dy < -1;
  const isFalling = player.dy > 2 && !player.onGround;

  // 1. Always increment the frame counter
  player.playerFrameIndex++;

  // 2. Handle Attack Logic
  if (player.isAttacking) {
    const comboFrame = Math.floor(player.playerFrameIndex / 4); // Kept faster speed (4)
    
    if (comboFrame >= 7) {
      player.isAttacking = false;
      player.playerFrameIndex = 0; // Reset for next use
    } else {
      if (comboFrame < 3) {
        img = sprites.attack;
        frame = comboFrame;
        frameCount = 3;
      } else {
        img = sprites.attack_end;
        frame = comboFrame - 3;
        frameCount = 4;
      }
    }
  }

  // 3. Handle Movement Logic
  // We use an "else if" here so we don't accidentally draw movement 
  // frames while the attack is still playing.
  if (!player.isAttacking) {
    if (isJumping) {
      img = sprites.jump;
      frameCount = 6;
    } else if (isFalling) {
      img = sprites.fall;
      frameCount = 4;
    } else if (isMoving) {
      img = sprites.run;
      frameCount = 10;
    } else {
      img = sprites.idle;
      frameCount = 10;
    }
    frame = Math.floor(player.playerFrameIndex / 8) % frameCount;
  }

  // 4. Drawing code...
  if (!img || img.width === 0) return;
  const frameWidth = img.width / frameCount;
  const frameHeight = img.height;
  const srcX = frame * frameWidth;

  // 2. Drawing Logic
  ctx.save();
  if (player.facing < 0) {
    ctx.translate(player.x + player.w, player.y + player.yOffset);
    ctx.scale(-1, 1);
    ctx.drawImage(img, srcX, 0, frameWidth, frameHeight, 0, 0, player.w, player.h);
  } else {
    ctx.drawImage(img, srcX, 0, frameWidth, frameHeight, player.x, player.y + player.yOffset, player.w, player.h);
  }
  ctx.restore();
}

    function handleKeyDown(event) {
      if (event.code === "Space" || event.code === "ArrowUp" || event.code === "KeyW") {
        event.preventDefault();
      }
      keys[event.code] = true;

if (event.code === "KeyX") {
    if (!player.isAttacking) { // Prevents re-triggering while already attacking
        player.isAttacking = true;
        player.playerFrameIndex = 0;
    }
}
  
      if (event.code === "Enter" && !running) {
        startGame();
      }
    }

    function handleKeyUp(event) {
      keys[event.code] = false;
    }

    window.addEventListener("keydown", handleKeyDown);

function spawnDamageText(x, y, text) {

    timer: 60 
  });
}

function checkAllMobsDefeated() {
  // Returns true if every mob has health <= 0
  return mobs.every(mob => mob.health <= 0);
}

function drawBooger(mob) {
    if (mob.removed) return; 

    // 1. Determine Animation State
    let animName = "moveRight";
    if (mob.isDead) {
        animName = "dying";
    } else if (mob.wasHit) {
        animName = (mob.facing === -1) ? "hitLeft" : "hitRight";
    } else {
        animName = (mob.facing === -1) ? "moveLeft" : "moveRight";
    }

    const anim = BOOGER_ANIM[animName];
    const sy = (anim.row - 1) * BOOGER_SIZE;
    const sx = Math.floor(Math.floor(mob.frame) * BOOGER_SIZE);

    const padding = 1;
ctx.drawImage(
    sprites.booger, 
    sx, sy + padding, BOOGER_SIZE, BOOGER_SIZE - padding, 
    mob.x, mob.y, 64, 64
);
    // 3. Update logic (Handle Frame + Removal Timer)
    if (mob.isDead) {
        // Increment frame only if we haven't reached the end
        if (mob.frame < anim.frames - 1) {
            mob.frame += 0.1;
        } else {
            // Reached the end: freeze frame and increment the death timer
            mob.frame = anim.frames - 1;
            mob.deathTimer++;
            
            // Mark for removal after timer
            if (mob.deathTimer > 60) {
                mob.removed = true;
            }
        }
    } else {
        // Normal looping animation
        mob.frame = (mob.frame + 0.1) % anim.frames;
    }
}

function checkWord() {

    const errorDisplay = document.getElementById("errorMessage");

    if (input === "summer") {
        window.location.href = "next_level.html"; 
    } else {
        // Show the message inline instead of an alert
        errorDisplay.textContent = "Incorrect word! Please try again.";
        
        // Optional: Clear the input so they can type again
        document.getElementById("passwordInput").value = ""; 
    }
}
